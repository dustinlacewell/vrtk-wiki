382  protected virtual IEnumerator TransitionTransform(TransformData source, TransformData target, bool dynamicDestination, Vector3 destinationScale, Quaternion destinationRotation, Vector3 destinationPosition)
383  {
384  Vector3 initialScale = target.Scale;
385  Quaternion initialRotation = target.Rotation;
386  Vector3 initalPosition = target.Position;
387 
388  float elapsedTime = 0f;
389  while (elapsedTime < TransitionDuration)
390  {
391  bool equalityCheck = dynamicDestination ?
392  ArePropertiesEqual(source.Position, target.Position, source.Rotation, target.Rotation, source.Scale, target.Scale) :
393  ArePropertiesEqual(initalPosition, destinationPosition, initialRotation, destinationRotation, initialScale, destinationScale);
394 
395  if (equalityCheck)
396  {
397  break;
398  }
399 
400  if (dynamicDestination)
401  {
402  destinationScale = Source.Scale;
403  destinationRotation = Source.Rotation;
404  destinationPosition = Source.Position;
405  }
406 
407  float lerpFrame = elapsedTime / TransitionDuration;
408  UpdateTransformProperties(target.Transform,
409  Vector3.Lerp(initialScale, destinationScale, lerpFrame),
410  Quaternion.Lerp(initialRotation, destinationRotation, lerpFrame),
411  Vector3.Lerp(initalPosition, destinationPosition, lerpFrame),
412  target.UseLocalValues);
413 
414  elapsedTime += Time.deltaTime;
415  yield return DelayInstruction;
416  }
417 
418  UpdateTransformProperties(target.Transform, destinationScale, destinationRotation, destinationPosition, target.UseLocalValues);
419  AfterTransformUpdated?.Invoke(eventData.Set(source, target));
420  }
